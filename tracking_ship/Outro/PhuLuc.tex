class UKF:
    """
    Unscented Kalman Filter cho state [x, y, v, psi, omega]
    """
    
    def __init__(self, dt, alpha=1e-3, beta=2.0, kappa=0.0):
        self.dt = dt
        self.n = 5  # [x,y,v,psi,omega]
        self.alpha = alpha
        self.beta = beta
        self.kappa = kappa
        self.lmbda = alpha**2 * (self.n + kappa) - self.n
        
        # Trong so cho sigma points
        self.Wm = np.full(2*self.n+1, 0.5/(self.n+self.lmbda))
        self.Wc = self.Wm.copy()
        self.Wm[0] = self.lmbda/(self.n+self.lmbda)
        self.Wc[0] = self.Wm[0] + (1 - alpha**2 + beta)
        
        # State va covariance
        self.x = np.zeros(self.n)
        self.P = np.eye(self.n)*1.0
        
        # Process noise va Measurement noises
        self.Q = np.diag([1.0, 1.0, 0.5, (np.deg2rad(1))**2, (np.deg2rad(0.1))**2])
        self.R_gps = np.diag([10.0**2, 10.0**2])
        self.R_imu = np.diag([0.5**2, (np.deg2rad(0.5))**2, (np.deg2rad(0.1))**2])
        self.R_odom = np.array([[0.3**2]])

    def generate_sigma_points(self, x, P):
        """Tao sigma points tu state va covariance"""
        n = self.n
        P = 0.5 * (P + P.T)  # Symmetrize
        scaleP = (n + self.lmbda) * P
        
        try:
            sqrtP = np.linalg.cholesky(scaleP)
        except np.linalg.LinAlgError:
            sqrtP = np.linalg.cholesky(scaleP + np.eye(n)*1e-9)
        
        sigmas = np.zeros((2*n+1, n))
        sigmas[0] = x
        for i in range(n):
            sigmas[i+1] = x + sqrtP[i]
            sigmas[n+i+1] = x - sqrtP[i]
        
        return sigmas

    def predict(self, dt=None):
        """Buoc predict cua UKF"""
        dt = dt or self.dt
        sigmas = self.generate_sigma_points(self.x, self.P)
        sigmas_f = np.array([self.f(s, dt) for s in sigmas])
        
        x_pred = np.sum(self.Wm[:,None] * sigmas_f, axis=0)
        P_pred = self.Q.copy()
        for i in range(sigmas_f.shape[0]):
            diff = sigmas_f[i] - x_pred
            P_pred += self.Wc[i] * np.outer(diff, diff)
        
        self.x, self.P = x_pred, P_pred

\subsubsection*{D.1. Cholesky Decomposition}

Cho ma trận đối xứng xác định dương $\mathbf{P} \in \mathbb{R}^{n \times n}$, Cholesky decomposition tìm ma trận tam giác dưới $\mathbf{L}$ sao cho:
\begin{equation}
\mathbf{P} = \mathbf{L}\mathbf{L}^T
\end{equation}

Algorithm:
\begin{algorithm}[H]
\For{$i = 0$ to $n-1$}{
    \For{$j = 0$ to $i$}{
        $sum = \sum_{k=0}^{j-1} L_{ik}L_{jk}$ \\
        \If{$i == j$}{
            $L_{ii} = \sqrt{P_{ii} - sum}$
        }
        \Else{
            $L_{ij} = \frac{1}{L_{jj}}(P_{ij} - sum)$
        }
    }
}
\end{algorithm}

\subsubsection*{D.2. Haversine Distance Formula}

Khoảng cách trên mặt cầu giữa hai điểm $(lat_1, lon_1)$ và $(lat_2, lon_2)$:

\begin{align}
a &= \sin^2\left(\frac{\Delta lat}{2}\right) + \cos(lat_1) \cdot \cos(lat_2) \cdot \sin^2\left(\frac{\Delta lon}{2}\right) \\
c &= 2 \cdot \arcsin(\sqrt{a}) \\
d &= R_{Earth} \cdot c
\end{align}

Trong đó $R_{Earth} = 6,371,000$ m.
